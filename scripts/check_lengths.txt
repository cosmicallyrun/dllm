import os
os.environ["HF_HOME"] = "/workspace/.hf_cache"
os.environ["HF_DATASETS_CACHE"] = "/workspace/.hf_cache/datasets"

from datasets import load_dataset
from transformers import AutoTokenizer

tok = AutoTokenizer.from_pretrained("/workspace/dllm/.models/a2d/Qwen3.5-0.8B")

def stats(name, lengths):
    u1k = sum(1 for l in lengths if l <= 1024)
    u2k = sum(1 for l in lengths if l <= 2048)
    med = sorted(lengths)[len(lengths)//2]
    print(f"\n{name} (n={len(lengths)})")
    print(f"  Under 1024: {u1k} ({100*u1k/len(lengths):.1f}%)")
    print(f"  Under 2048: {u2k} ({100*u2k/len(lengths):.1f}%)")
    print(f"  Median: {med} tokens")

# s1K
ds = load_dataset("simplescaling/s1K", split="train")
lengths = [len(tok.encode(ex["question"] + " " + (ex["thinking_trajectories"][0] if ex["thinking_trajectories"] else "") + " " + (ex["attempt"] or ""))) for ex in ds]
stats("simplescaling/s1K", lengths)

# Hermes thinking
ds = load_dataset("Jofthomas/hermes-function-calling-thinking-V1", split="train[:500]")
lengths = [len(tok.encode(" ".join(t.get("value","") for t in ex.get("conversations",[])))) for ex in ds]
stats("Jofthomas/hermes-function-calling-thinking-V1", lengths)

# Hermes plain
ds = load_dataset("NousResearch/hermes-function-calling-v1", split="train[:500]")

stats("NousResearch/hermes-function-calling-v1", lengths)

# Alpaca
ds = load_dataset("tatsu-lab/alpaca", split="train[:500]")

stats("tatsu-lab/alpaca", lengths)

# UltraChat
ds = load_dataset("HuggingFaceH4/ultrachat_200k", split="train_gen[:500]")

stats("HuggingFaceH4/ultrachat_200k", lengths)
